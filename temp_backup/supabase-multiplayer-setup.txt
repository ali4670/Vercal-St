-- MULTIPLAYER & SOCIAL SETUP FOR ST-COMPANY X-O
-- Run this in your Supabase SQL Editor

-- 1. Ensure profiles are searchable (already exists, but adding a helpful index)
CREATE INDEX IF NOT EXISTS idx_profiles_username ON profiles USING gin (username lgm_ops);
-- Note: pg_trgm extension might be needed for fuzzy search: CREATE EXTENSION IF NOT EXISTS pg_trgm;

-- 2. Create Games table for online play
CREATE TABLE IF NOT EXISTS games (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  player_x UUID REFERENCES profiles(id) NOT NULL,
  player_o UUID REFERENCES profiles(id), -- Null means waiting for opponent
  board JSONB DEFAULT '[null, null, null, null, null, null, null, null, null]'::jsonb,
  current_turn TEXT DEFAULT 'X',
  winner TEXT, -- 'X', 'O', or 'draw'
  status TEXT DEFAULT 'pending', -- 'pending', 'active', 'completed', 'cancelled'
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now())
);

-- 3. Enable Realtime for the games table
ALTER PUBLICATION supabase_realtime ADD TABLE games;

-- 4. Security Policies for Games
ALTER TABLE games ENABLE ROW LEVEL SECURITY;

-- Anyone can see games they are part of
CREATE POLICY "View own games" ON games 
FOR SELECT USING (auth.uid() = player_x OR auth.uid() = player_o);

-- Anyone can create a game
CREATE POLICY "Create games" ON games 
FOR INSERT WITH CHECK (auth.uid() = player_x);

-- Players can update their own games
CREATE POLICY "Update own games" ON games 
FOR UPDATE USING (auth.uid() = player_x OR auth.uid() = player_o);
